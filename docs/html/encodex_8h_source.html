104/****************************************************************
105 * @brief Parses ASN.1 encoded data and returns metadata into an array of structs.
106 * @note This function is recursive for any element of @b constructed form.
107 * @note For DER-formatted RSA public keys, you will need to call this function twice to
108 * unpack the modulus and exponent. The second time should be on the
109 * @b ASN1_BITSTRING that encodes the modulus
110 * and public exponent. See the asn1\_decode demo for details.
111 * @param asn1_data Pointer to ASN.1-encoded data.
112 * @param len The length of the encoded data.
113 * @param objs Pointer to an array of @b asn1_obj_t structs to fill with decoded data.
114 * @param iter_count Maximum number of ASN.1 elements to process before returning.
115 * @returns The number of objects returned by the parser. Zero indicates an error.
116 */
117size_t asn1_decode(void *asn1_data, size_t len, asn1_obj_t *objs, size_t iter_count);
